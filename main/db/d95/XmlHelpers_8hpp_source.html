77 template <std::size_t N>
78 Eigen::Matrix<double, N, 1> toVectorNd(const std::string& str) {
79  Eigen::Matrix<double, N, 1> ret = Eigen::Matrix<double, N, 1>::Zero();
80 
81  std::vector<std::string> pieces;
82  std::string trimedStr = boost::trim_copy(str);
83  boost::split(
84  pieces, trimedStr, boost::is_any_of(" "), boost::token_compress_on);
85  std::size_t sizeToRead = std::min(N, pieces.size());
86  if (pieces.size() < N) {
87  dterr << "Failed to read a vector because the dimension '" << pieces.size()
88  << "' is less than the expectation '" << N << "'.\n";
89  } else if (pieces.size() > N) {
90  dterr << "Failed to read a vector because the dimension '" << pieces.size()
91  << "' is greater than the expectation '" << N << "'.\n";
92  }
93 
94  for (std::size_t i = 0; i < sizeToRead; ++i) {
95  if (pieces[i] != "") {
96  try {
97  ret(i) = boost::lexical_cast<double>(pieces[i].c_str());
98  } catch (boost::bad_lexical_cast& e) {
99  dterr << "value [" << pieces[i]
100  << "] is not a valid double for Eigen::Vector" << N << "d[" << i
101  << "]: " << e.what() << "\n";
102  }
103  }
104  }
105 
106  return ret;
107 }

208 template <typename ElementType>
209 class TemplatedElementEnumerator {
210 protected:
211  using ElementPtr = ElementType*;
212  using ElementRef = ElementType&;
213 
214 public:
216  TemplatedElementEnumerator(
217  ElementPtr parentElement, const std::string& childElementName)
218  : mParentElement(parentElement),
219  mChildElementName(childElementName),
220  mCurrentElement(nullptr) {
221  }
222 
224  ~TemplatedElementEnumerator() {
225  }
226 
229  bool next() {
230  if (!mParentElement)
231  return false;
232 
233  if (mCurrentElement) {
234  mCurrentElement
235  = mCurrentElement->NextSiblingElement(mChildElementName.c_str());
236  } else {
237  mCurrentElement
238  = mParentElement->FirstChildElement(mChildElementName.c_str());
239  }
240 
241  if (!valid())
242  mParentElement = nullptr;
243 
244  return valid();
245  }
246 
248  ElementPtr get() const {
249  return mCurrentElement;
250  }
251 
253  ElementPtr operator->() const {
254  return mCurrentElement;
255  }
256 
258  ElementRef operator*() const {
259  return *mCurrentElement;
260  }
261 
263  bool operator==(const TemplatedElementEnumerator<ElementType>& rhs) const {
264  // If they point at the same node, then the names must match
265  return (this->mParentElement == rhs.mParentElement)
266  && (this->mCurrentElement == rhs.mCurrentElement)
267  && (this->mCurrentElement != nullptr
268  || (this->mChildElementName == rhs.mChildElementName));
269  }
270 
272  TemplatedElementEnumerator<ElementType>& operator=(
273  const TemplatedElementEnumerator<ElementType>& rhs) {
274  this->mParentElement = rhs.mParentElement;
275  this->mChildElementName = rhs.mChildElementName;
276  this->mCurrentElement = rhs.mCurrentElement;
277 
278  return *this;
279  }
280 
281 private:
283  bool valid() const {
284  return mCurrentElement != nullptr;
285  }
286 
287 private:
289  ElementPtr mParentElement;
290 
292  std::string mChildElementName;
293 
295  ElementPtr mCurrentElement;
296 };
